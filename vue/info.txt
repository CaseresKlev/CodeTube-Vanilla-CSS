<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8"/>
    <title>Vue.js Tutorial</title>
    <link href="https://cdn.jsdelivr.net/npm/bootstrap@4.1.1/dist/css/bootstrap.css" rel="stylesheet"/>
    <style>
    
    </style>
    <script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/sweetalert"></script>
</head>
<body>
    
    <div id="app">
        <div class="container">
            <h1>{{ title }}</h1>
            <div class="row">
                <div class="col-md-5">
                    <select size="30" id="list1" class="form-control" multiple>
                        <option v-for="item1 in list1">{{ item1 }}</option>
                    </select>
                </div>
                <div class="col-md-1">
                    <button class="btn btn-primary mb-2 btn-block" @click="allToRight">&raquo;</button>
                    <button class="btn btn-primary mb-2 btn-block" @click="oneToRight">&rsaquo;</button>
                    <button class="btn btn-primary mb-2 btn-block" @click="oneToLeft">&lsaquo;</button>
                    <button class="btn btn-primary mb-2 btn-block" @click="allToLeft">&laquo;</button>
                </div>
                <div class="col-md-5">
                    <select size="30" id="list2" class="form-control" multiple>
                        <option v-for="item2 in list2">{{ item2 }}</option>
                    </select>
                </div>
            </div>
        </div>
    </div>
    
    <script>
        var app = new Vue({
            el: '#app',
            data: {
                title: 'Dual List Vue.js',
                list1: [ 'Tedir', 'Ghazali', 'Alexander', 'Murad', 'Bukhari', 'Boyhaki', 'Nurjannah', 'Fatma', 'Bakri' ],
                list2: [ 'Razali', 'Rizal' ]
            },
            methods: {
                oneToRight: function () {
                    var select1 = document.getElementById('list1').value
                    if(select1 != ""){
                        this.list2.push(select1)
                        var del = this.list1.indexOf(select1)
                        this.list1.splice(del, 1)
                    }
                },
                oneToLeft: function () {
                    var select2 = document.getElementById('list2').value
                    if(select2 != ""){
                        this.list1.push(select2)
                        var del = this.list2.indexOf(select2)
                        this.list2.splice(del, 1)
                    }
                },
                allToRight: function () {
                    for(var i = 0; i < this.list1.length; i++){
                        this.list2.push(this.list1[i])
                        var del = this.list1.indexOf(this.list1[i])
                        this.list1.splice(del, 1)
                    }
                },
                allToLeft: function () {
                    for(var i = 0; i < this.list2.length; i++){
                        this.list1.push(this.list2[i])
                        var del = this.list2.indexOf(this.list2[i])
                        this.list2.splice(del, 1)
                    }
                }
            }
        })
    </script>
